#### 1. 取最近一个不为空的值填充
```sql
WITH mian_data AS (
    SELECT
        c0,
        c1,
        IF(c1 = '处置', NULL, c2) AS c2
    FROM
        VALUES
         (1,'处置','1')
        ,(2,'处置','2')
        ,(3,'处置','3')
        ,(4,'处置','4')
        ,(5,'提交','A')
        ,(6,'处置','5')
        ,(7,'提交','b')
    AS a(c0,c1,c2)
)
SELECT 
    *,
    FIRST_VALUE(c2, True) over(order by c0 ROWS BETWEEN CURRENT ROW AND UNBOUNDED FOLLOWING)
FROM 
    mian_data 
```
